class HierarchicalSharedRoutedMoE(nn.Module):
    """åˆ†å±‚å…±äº«è·¯ç”±MoE"""
    def __init__(self, in_channels, num_shared=2, num_routed=4):

        self.num_shared = num_shared
        self.num_routed = num_routed

        # å…±äº«ä¸“å®¶ (å§‹ç»ˆæ´»è·ƒ)
        self.shared_experts = nn.ModuleList([
            nn.Sequential(
                nn.Conv2d(in_channels, in_channels, 1),
                nn.GELU(),
                nn.Conv2d(in_channels, in_channels, 1)
            ) for _ in range(num_shared)

        # è·¯ç”±ä¸“å®¶ (ç¨€ç–æ¿€æ´»)
        self.routed_experts = nn.ModuleList([
            nn.Sequential(
                nn.Conv2d(in_channels, in_channels, 1),
                nn.GELU(),
                nn.Conv2d(in_channels, in_channels, 1)
            ) for _ in range(num_routed)
        ])

        self.router = nn.Conv2d(in_channels, num_routed, 1)

        # è´Ÿè½½å‡è¡¡
        self.register_buffer('expert_counts', torch.zeros(num_routed))

    def forward(self, x, lambda_bal=0.1):
        """
        x: (B, C, H, W)
        """
        B, C, H, W = x.shape

        # 1. å…±äº«ä¸“å®¶è¾“å‡º
        shared_out = 0
        for expert in self.shared_experts:
            shared_out = shared_out + expert(x)
        shared_out = shared_out / self.num_shared

        # 2. è·¯ç”±
        logits = self.router(x)  # (B, Nr, H, W)
        gates = F.softmax(logits, dim=1)  # (B, Nr, H, W)

        # 3. Top-1é€‰æ‹©
        gate_max, expert_indices = gates.max(dim=1, keepdim=True)  # (B, 1, H, W)
        mask = torch.zeros_like(gates).scatter_(1, expert_indices, 1)

        # 4. è·¯ç”±ä¸“å®¶è¾“å‡º
        routed_out = 0
        for e_idx, expert in enumerate(self.routed_experts):
            expert_mask = mask[:, e_idx, :, :]  # (B, 1, H, W)
            expert_out = expert(x) * expert_mask
            routed_out = routed_out + expert_out

        # 5. æ›´æ–°è´Ÿè½½ç»Ÿè®¡ (ç”¨äºè®­ç»ƒæ—¶çš„è´Ÿè½½å‡è¡¡æŸå¤±)
        if self.training:
            with torch.no_grad():
                expert_counts = mask.mean(dim=[0, 2, 3])  # (Nr,)
                self.expert_counts = 0.9 * self.expert_counts + 0.1 * expert_counts

        # 6. ç»„åˆè¾“å‡º
        output = shared_out + routed_out

        return output, gates

    def load_balance_loss(self, gates):
        """è®¡ç®—è´Ÿè½½å‡è¡¡æŸå¤±"""
        # gates: (B, Nr, H, W)
        B, Nr, H, W = gates.shape
