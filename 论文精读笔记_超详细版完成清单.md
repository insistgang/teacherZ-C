def manhole_detection_comprehensive(image_or_video):
    """
    终极井盖检测方案（基于26篇精读论文）
    """
    # ===== 预处理阶段 =====
    # [2-01] Split Bregman去噪
    u_clean = split_bregman_rof(image, lambda_param=0.1)
    
    # [2-08] 框架增强
    coeffs = frame_decomposition(u_clean, J=3)
    u_enhanced = enhance_circle_features(coeffs)
    
    # ===== 分割阶段 =====
    # [2-03] SLaT粗分割
    seg_coarse = slat_segmentation(u_enhanced, K=3)
    
    # [2-04] 联合精修
    seg_refined = joint_refinement(u_enhanced, seg_coarse)
    
    # [2-05] 比例约束过滤
    seg_filtered = proportion_constraint_filter(seg_refined, p_manhole=(0.01, 0.1))
    
    # ===== 几何精修阶段 =====
    # [2-15] 水平集圆形拟合
    phi = circular_levelset_refinement(seg_filtered)
    
    # [2-09] 曲率约束精修
    phi_smooth = curvature_constrained_refinement(phi)
    
    # ===== 深度学习验证 =====
    # [2-20] U-Net验证
    if unet_confidence(phi_smooth) < 0.9:
        phi_final = unet_correction(phi_smooth)
    else:
        phi_final = phi_smooth
    
    # ===== 输出 =====
    center, radius = extract_circle_from_sdf(phi_final)
    confidence = compute_detection_confidence(phi_final)
    
    return center, radius, confidence
