import numpy as np

from sklearn.cluster import KMeans

def sat_segmentation(f, lambda_=1.0, mu=1.0, K=2, max_iter=200):
    """
    SaTåˆ†å‰²ç®—æ³•

    å‚æ•°:
        f: è¾“å…¥å›¾åƒ
        lambda_: å¹³æ»‘æƒé‡
        mu: TVæƒé‡
        K: åˆ†å‰²ç±»åˆ«æ•°
        max_iter: æœ€å¤§è¿­ä»£æ¬¡æ•°
    """
    # Split-Bregmanæ±‚è§£å¹³æ»‘è§£g
    g = split_bregman_sat(f, lambda_, mu, max_iter)

    # å½’ä¸€åŒ–
    g_norm = (g - g.min()) / (g.max() - g.min())

def split_bregman_sat(f, lambda_, mu, max_iter):
    """Split-Bregmanæ±‚è§£SaTæ¨¡å‹"""
    sigma = 2.0

    g = f.copy()
    dx = np.zeros_like(f)
    dy = np.zeros_like(f)
    bx = np.zeros_like(f)
    by = np.zeros_like(f)

    for k in range(max_iter):
        g_old = g.copy()

        # g-å­é—®é¢˜ï¼ˆFFTæ±‚è§£ï¼‰
        rhs = lambda_*f + sigma*div(dx-bx, dy-by)
        g = fft_solve(rhs, lambda_, mu, sigma)

        # d-å­é—®é¢˜
        sx = grad_x(g) + bx
        sy = grad_y(g) + by
        s = np.sqrt(sx**2 + sy**2) + 1e-10
        dx = np.maximum(s - 1/sigma, 0) * sx / s
        dy = np.maximum(s - 1/sigma, 0) * sy / s

        # Bregmanæ›´æ–°
        bx = bx + grad_x(g) - dx
        by = by + grad_y(g) - dy

        # æ”¶æ•›æ£€æŸ¥
        if np.linalg.norm(g-g_old) / np.linalg.norm(g) < 1e-4:
            break

    return g

def slat_segmentation(f_rgb, lambda_=1.0, mu=1.0, K=3):
    """SLaTå½©è‰²å›¾åƒåˆ†å‰²"""
    from skimage.color import rgb2lab

    # ç¬¬ä¸€é˜¶æ®µï¼šå¹³æ»‘RGB

    # ç¬¬äºŒé˜¶æ®µï¼šæå‡åˆ°Lab
    g_lab = rgb2lab(g_rgb)

    g_flat = g_6d.reshape(-1, 6)

    labels = kmeans.labels_.reshape(h, w)

    return labels
