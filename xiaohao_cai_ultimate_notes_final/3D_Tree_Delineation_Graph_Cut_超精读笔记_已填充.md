import numpy as np

from scipy.sparse.csgraph import connected_components
from sklearn.neighbors import kneighbors_graph
from sklearn.cluster import KMeans

def mcrc_delineation(points, chm, hyperspectral=None, C=None,
                      sigma_xy=2, sigma_z=3, sigma_fts=0.005,
                      kappa=0.1, max_depth=5, min_size=50):
    """

        C: èšç±»æ•° (è‡ªåŠ¨æ£€æµ‹å¦‚æœä¸ºNone)

    for i, top in enumerate(tree_tops):

        distances = np.linalg.norm(points - top, axis=1)
        S[:, i] = np.exp(-distances**2 / (2 * sigma_xy**2))

    # 3. æ„å»ºkè¿‘é‚»å›¾
    k = min(30, N - 1)
    W = kneighbors_graph(points, k, mode='distance')

    # 4. è®¡ç®—èåˆæƒé‡
    if hyperspectral is not None:
        # æ·»åŠ å…‰è°±ç›¸ä¼¼æ€§
        spectral_dist = np.linalg.norm(
            hyperspectral[:, None, :] - hyperspectral[None, :, :], axis=2

        W.data *= np.exp(-spectral_dist[W.row, W.col]**2 / sigma_fts**2)

    # 5. å¤šåº¦çŸ©é˜µ
    D = np.diag(W.sum(axis=1).A1)

            X[:, i] *= -1

def recursive_cut(points, labels, W, D, depth=0,
                  max_depth=5, min_size=50):
    """
    é€’å½’Normalized Cut
    """

    result = labels.copy()

    for label in unique_labels:
        mask = labels == label
        if np.sum(mask) < min_size or depth >= max_depth:
            continue

        # æå–å­å›¾
        sub_points = points[mask]
        sub_indices = np.where(mask)[0]

        # æ„å»ºå­å›¾æƒé‡
        sub_W = W[np.ix_(sub_indices, sub_indices)]

        # äºŒå…ƒNormalized Cut
        sub_D = np.diag(sub_W.sum(axis=1).A1)

        # æ±‚è§£ç¬¬äºŒå°ç‰¹å¾å‘é‡
        eigvals, eigvecs = np.linalg.eigh(sub_L)
        fiedler = eigvecs[:, 1]

        threshold = np.median(fiedler)
        binary_labels = (fiedler > threshold).astype(int)

        # æ›´æ–°ç»“æœ
        new_labels = np.where(binary_labels == 0,
                             label,
                             label + len(unique_labels))
        result[sub_indices] = new_labels

        # é€’å½’å¤„ç†å­èšç±»
        result = recursive_cut(points, result, W, D,
                               depth=depth+1,
                               max_depth=max_depth,
                               min_size=min_size)

    return result
