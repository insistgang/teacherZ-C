def TV_gradient(image):

    # å‰å‘å·®åˆ†

    grad[1:, :] += grad_y[:-1, :]

    return grad

def proximal_tv(image, lambda_tv, n_iter=100):
    """TVè¿‘ç«¯ç®—å­ï¼ˆä½¿ç”¨Chambolle-Pockï¼‰"""

        grad_x = np.array([np.gradient(x, axis=1), np.gradient(x, axis=0)])
        u_new = u + 0.5 * grad_x

    return x
